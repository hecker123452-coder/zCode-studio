    55тЖТ    55тЖТfunction parseAgentFiles(reply: string, defaultFileName: string): { fileName: string, content: string }[] {
    56тЖТ    56тЖТ  const actions: { fileName: string, content: string }[] = []
    57тЖТ    57тЖТ  // Match code blocks with filename: ```lang:filename
    58тЖТ    58тЖТ  const fileRegex = /```[\w]*:(\S+)\n([\s\S]*?)```/g
    59тЖТ    59тЖТ  let match
    60тЖТ    60тЖТ  while ((match = fileRegex.exec(reply)) !== null) {
    61тЖТ    61тЖТ    actions.push({ fileName: match[1], content: match[2] })
    62тЖТ    62тЖТ  }
    63тЖТ    63тЖТ  // If no filename blocks, try regular code blocks
    64тЖТ    64тЖТ  if (actions.length === 0) {
    65тЖТ    65тЖТ    const codeRegex = /```[\w]*\n([\s\S]*?)```/g
    66тЖТ    66тЖТ    while ((match = codeRegex.exec(reply)) !== null) {
    67тЖТ    67тЖТ      if (match[1].split('\n').length > 3) {
    68тЖТ    68тЖТ        actions.push({ fileName: defaultFileName, content: match[1] })
    69тЖТ    69тЖТ        break
    70тЖТ    70тЖТ      }
    71тЖТ    71тЖТ    }
    72тЖТ    72тЖТ  }
    73тЖТ    73тЖТ  return actions
    74тЖТ    74тЖТ}

    76тЖТ    76тЖТ// Extract thinking steps from reply (bullet points before code blocks)
    77тЖТ    77тЖТfunction parseThinkingSteps(reply: string): string[] {
    78тЖТ    78тЖТ  const beforeCode = reply.split(/```/)[0] || ''
    79тЖТ    79тЖТ  const lines = beforeCode.split('\n').filter(l => l.trim().startsWith('-') || l.trim().startsWith('тАв'))
    80тЖТ    80тЖТ  return lines.map(l => l.trim().replace(/^[-тАв]\s*/, '')).filter(l => l.length > 0)
    81тЖТ    81тЖТ}
